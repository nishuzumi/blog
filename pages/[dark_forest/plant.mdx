# 星球

## 星球代码

Path: eth/contracts/DFTypes.sol

```sol
struct Planet {
    address owner;                    // 星球的当前所有者的以太坊地址
    uint256 range;                    // 从该星球可以发送舰队到达的最远距离
    uint256 speed;                    // 从该星球发送的舰队的移动速度
    uint256 defense;                  // 星球的防御力，可能影响战斗结果
    uint256 population;               // 星球上的当前人口或能量
    uint256 populationCap;            // 星球的人口或能量上限
    uint256 populationGrowth;         // 星球人口或能量的增长率
    uint256 silverCap;                // 星球上银币的上限
    uint256 silverGrowth;             // 星球上银币的增长率
    uint256 silver;                   // 星球上的当前银币数量
    uint256 planetLevel;              // 星球的等级，可能影响其其他属性
    PlanetType planetType;            // 星球的类型，例如气态、岩石等
    bool isHomePlanet;                // 标记该星球是否是玩家的主星
    bool isInitialized;               // 标记该星球是否已被初始化或被玩家发现
    uint256 createdAt;                // 星球创建的时间或区块
    uint256 lastUpdated;              // 星球上次更新的时间或区块
    uint256 perlin;                   // 可能与星球的随机生成或地形有关的噪音值
    SpaceType spaceType;              // 星球所在的空间类型
    uint256 upgradeState0;            // 星球的第一个升级状态
    uint256 upgradeState1;            // 星球的第二个升级状态
    uint256 upgradeState2;            // 星球的第三个升级状态
    uint256 hatLevel;                 // 星球的“帽子”等级，可能是一种装饰或标志
    bool hasTriedFindingArtifact;     // 标记玩家是否尝试在该星球上找到遗物
    uint256 prospectedBlockNumber;    // 玩家上次在该星球上勘探的区块号
    bool destroyed;                   // 标记该星球是否已被摧毁
    uint256 spaceJunk;                // 星球上的太空垃圾数量
    uint256 pausers;                  // 星球上的“暂停”数量或状态
    uint256 energyGroDoublers;        // 与星球能量增长加倍器有关
    uint256 silverGroDoublers;        // 与星球银币增长加倍器有关
    address invader;                  // 当前正在入侵星球的玩家地址
    uint256 invadeStartBlock;         // 入侵开始的区块号
    address capturer;                 // 最后一次成功捕获星球的玩家地址
    uint256 locationId;               // 星球的唯一位置ID
}
```
